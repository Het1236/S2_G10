\section{Opening Context (Real-Life Connection)}
The lecture is motivated by the Amazon Black Friday Sale 2026, where massive traffic spikes are expected. 

\begin{itemize}
    \item Systems must handle extreme (rare) events, not just average load.
    \item The key engineering problem: Predict probability of system failure due to unusually high traffic (tail events).
\end{itemize}

\textbf{From page 5 visualization:}
Mean load is stable ($\sim 50$ requests/min), but rare spikes exceed threshold ($k = 60$) causing crashes. \\
Empirical crash rate shown: 8.7\% (87 crashes / 1000 mins). 

\section{Definitions and Notation}
\subsection{Random Variable and Tail Event}
Let:
\begin{itemize}
    \item $X$: number of requests in a given time window
    \item $\mu = E[X]$: mean load
    \item $\sigma^2 = \text{Var}(X)$: variance
\end{itemize}

\textbf{Tail Event (as used in lecture):}
\[ P(X \ge k) \]
where $k$ is a critical threshold (system capacity limit).

\subsection{Problem Statement (Implicit in Lecture)}
\textbf{Given:} Mean (and possibly variance / distribution) \\
\textbf{Goal:} Estimate $P(X \ge k)$ \\
\textbf{Constraint:} Exact distribution often unknown or hard to compute.

\section{Key Idea: Probability Bounds}
\textbf{Statement (as conveyed in lecture):} \\
Instead of computing exact probability, use upper bounds on $P(X \ge k)$ based on available information.

\vspace{0.3cm}
\noindent\textbf{Hierarchy of Bounds (from page 6 pyramid)}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|c|l|l|l|}
\hline
\textbf{Level} & \textbf{Method} & \textbf{Requires} & \textbf{Tightness} \\
\hline
1 & Markov & Mean $\mu$ & Loose \\
2 & Chebyshev & Mean + Variance & Moderate \\

\section{Theorem 1: Markov’s Inequality}
\subsection{Statement}
For a non-negative random variable $X$ and $a > 0$:
\[ P(X \ge a) \le \frac{E[X]}{a} \]

\subsection{Conditions}
\begin{itemize}
    \item $X \ge 0$
    \item Only requires mean $\mu$
\end{itemize}

\subsection{Proof Sketch (as implied in lecture)}
\textbf{Step 1:} Observe that for $X \ge a$,
\[ X \ge a \cdot \mathbf{1}_{X \ge a} \]
\textbf{Step 2:} Take expectation:
\[ E[X] \ge a \cdot P(X \ge a) \]
\textbf{Step 3:} Rearranging:
\[ P(X \ge a) \le \frac{E[X]}{a} \]

\subsection{Interpretation (from lecture)}
\begin{itemize}
    \item Uses only mean $\rightarrow$ very weak information.
    \item \textbf{From page 7 graph:} Bound decays as $1/k$ (linear decay in log-scale).
    \item Overestimates heavily. Example shown: Markov $\approx 0.714$ vs Exact $\approx 0.004$.
\end{itemize}

\section{Theorem 2: Chebyshev’s Inequality}
\subsection{Statement}
\[ P(|X - \mu| \ge a) \le \frac{\sigma^2}{a^2} \]
Equivalent tail form:
\[ P(X \ge \mu + a) \le \frac{\sigma^2}{a^2} \]

\subsection{Conditions}
Requires:
\begin{itemize}
    \item Mean $\mu$
    \item Variance $\sigma^2$
\end{itemize}

\subsection{Proof Sketch (conceptual flow)}
Derived using Markov inequality applied to $(X - \mu)^2$.
\begin{itemize}
    \item Let $Y = (X - \mu)^2$
    \item Apply Markov: $P(Y \ge a^2) \le \frac{E[Y]}{a^2}$
    \item Since $E[Y] = \sigma^2$:
    \[ P(|X - \mu| \ge a) \le \frac{\sigma^2}{a^2} \]
\end{itemize}

\subsection{Interpretation}
\begin{itemize}
    \item \textbf{From page 8 graph:} Decay is polynomial ($1/k^2$).
    \item Tighter than Markov but still loose.
\end{itemize}

\section{Theorem 3: Chernoff Bound}
\subsection{Statement (as shown in lecture)}
\[ P(X \ge a) \le \min_{t > 0} e^{-ta} M_X(t) \]
where:
\[ M_X(t) = E[e^{tX}] \quad \text{(Moment Generating Function)} \]

\subsection{Conditions}
\begin{itemize}
    \item Requires full distribution (MGF exists)
    \item $t > 0$
\end{itemize}

\subsection{Proof Sketch (as structured in page 18 blueprint)}
\textbf{Step 1:} Start from:
\[ P(X \ge a) = P(e^{tX} \ge e^{ta}) \]
\textbf{Step 2:} Apply Markov:
\[ P(X \ge a) \le \frac{E[e^{tX}]}{e^{ta}} \]
\textbf{Step 3:} Substitute MGF:
\[ = e^{-ta} M_X(t) \]
\textbf{Step 4:} Optimize over $t > 0$.

\subsection{Interpretation}
\begin{itemize}
    \item \textbf{From page 9 graph:} Exponential decay $e^{-k}$.
    \item Tightest bound among the three.
    \item Example: Chernoff $\approx 0.391$ vs Exact $\approx 0.092$ (close).
\end{itemize}

\section{What These Bounds Actually Do (From Lecture)}
Compute worst-case upper bound on failure probability $P(X \ge k)$. \\
\textbf{Used for:}
\begin{itemize}
    \item System reliability guarantees
    \item Capacity planning
    \item SLA validation
\end{itemize}

\section{Worked Case Study: Black Friday SLA Breach}
\subsection{Problem Setup (from page 11)}
\begin{itemize}
    \item \textbf{Traffic:} $X \sim \text{Poisson}(\lambda = 500)$
    \item \textbf{Threshold:} $k = 600$
    \item \textbf{Goal:} Compute $P(X \ge 600)$
\end{itemize}

\subsection{Results (from page 12 comparison)}
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|l|l|}
\hline
\textbf{Method} & \textbf{Estimated Probability} \\
\hline
Exact & $\sim 1.4\%$ \\
Chernoff & $\sim 8.8\%$ \\
Chebyshev & $\sim 20\%$ \\
Markov & $\sim 90\%$ \\
\hline
\end{tabular}
\end{center}

\subsection{Interpretation}
\begin{itemize}
    \item Markov $\rightarrow$ catastrophically loose
    \item Chebyshev $\rightarrow$ moderate
    \item Chernoff $\rightarrow$ closest to reality
\end{itemize}

\section{Engineering Insight (Critical from Lecture)}
\textbf{From pages 13–14:} \\
Waste Multiplier:
\[ \text{Waste} = \frac{\text{Bound Estimate}}{\text{Exact Probability}} \]

\textbf{Consequence:} Loose bounds $\rightarrow$ over-provisioning $\rightarrow$ financial loss.
\begin{itemize}
    \item Markov leads to $1000\times$ overestimation.
    \item Chernoff $\sim 10\times$ (acceptable).
\end{itemize}

    \item Apply Chernoff bound

\section{Summary \& Key Takeaways}
\begin{itemize}
    \item Tail probability $P(X \ge k)$ drives system failure analysis.
    \item Three bounds differ in information vs accuracy tradeoff:
    \begin{itemize}
        \item Markov $\rightarrow$ minimal info, poor accuracy
        \item Chebyshev $\rightarrow$ moderate
        \item Chernoff $\rightarrow$ best
    \end{itemize}
    \item Exponential decay (Chernoff) matches real-world behavior.
    \item Poor math $\rightarrow$ huge financial loss in systems engineering.
\end{itemize}
